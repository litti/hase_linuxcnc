# IP-Adresse auf Netzwerkinterface binden
# MESA läuft auf 192.168.1.121
apt install ethtool
ip a #Namen des Interfaces herausfinden
vi /etc/network/interfaces.d/mesa

## diesen inhalt da rein
uto enp87s0
iface enp87s0 inet static
  address 192.168.1.100/24
  mtu 1500
  pre-up /usr/sbin/ethtool -s enp87s0 speed 1000 duplex full autoneg off
  pre-up /usr/sbin/ethtool -K enp87s0 rx off tx off sg off tso off gso off gro off lro off
  pre-up /usr/sbin/ethtool -G enp87s0 rx 1024 tx 1024

  up   ip address add 192.168.100.11/24 dev $IFACE
  down ip address del 192.168.100.11/24 dev $IFACE

# !!! wenn es zu timing issues kommt dann speed 100 und die lokale adresse 192.168.100.... entfernen
